// logger.js
const pino = require('pino');

const logger = pino({

**src/provider.js anpassen:**

1. Imports am Anfang ersetzen/ergänzen:
```js
const logger = require('./logger');
const { AXIOS_TIMEOUT_MS, DEFAULT_LIMIT, MAX_LIMIT, CACHE_DB_PATH } = require('./config');
```

2. `const dbPath = process.env.CACHE_DB || path.join(...)` ersetzen durch:
```js
const dbPath = CACHE_DB_PATH;
```

3. Alle `console.log(...)` in provider.js ersetzen:
- `console.log('[cache] SQLite cache initialized at ...')` → `logger.info({ path: dbPath }, 'SQLite cache initialized')`
- `console.error('[cache] Failed to initialize SQLite cache ...')` → `logger.error({ err: err.message, path: dbPath }, 'Failed to initialize SQLite cache — running without cache')`
- `console.log('[cache] HIT for ...')` → `logger.debug({ key: cacheKey }, 'cache hit')`
- `console.error('[cache] Read error for ...')` → `logger.warn({ key: cacheKey, err: err.message }, 'cache read error — falling back to API')`
- `console.log('Found ${books.length} books ...')` → `logger.info({ count: books.length, query: formattedQuery }, 'search results received')`
- `console.log('Retrying without author: ...')` → `logger.info({ query: queryOnly }, 'retrying without author')`
- `console.log('Retry found ${books.length} books')` → `logger.info({ count: books.length }, 'retry results received')`
- `console.log('[cache] WRITE for ...')` → `logger.debug({ key: cacheKey }, 'cache write')`
- `console.error('[cache] Write error for ...')` → `logger.warn({ key: cacheKey, err: err.message }, 'cache write error')`
- `console.error('Error searching books:', error.message)` → `logger.error({ err: error.message, query: formattedQuery }, 'Storytel API request failed')`

4. `timeout: 30000` in fetchFromApi ersetzen durch `timeout: AXIOS_TIMEOUT_MS`.

5. `searchBooks(query, author = '', locale, type = 'all', limit = 20)` → `limit = DEFAULT_LIMIT`. `Math.min(Math.max(limit, 1), 50)` → `Math.min(Math.max(limit, 1), MAX_LIMIT)`.
