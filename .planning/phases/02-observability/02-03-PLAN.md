```js
/**
 * Returns current cache status for health endpoint
 * @returns {{ available: boolean, size: number|null }}
 */
function getDbStatus() {
    if (!db) return { available: false, size: null };
    try {
        const row = db.prepare('SELECT COUNT(*) as count FROM search_cache').get();
        return { available: true, size: row.count };
    } catch (err) {
        return { available: false, size: null };
    }
}

/**
 * Closes the SQLite database connection gracefully
 */
function closeDb() {
    if (db) {
        try {
            db.close();
            db = null;
        } catch (err) {
            // best-effort close
        }
    }
}
```

`module.exports` am Ende anpassen:

```js
module.exports = { StorytelProvider, StorytelApiError, getDbStatus, closeDb };
```

**4. Graceful Shutdown Handler nach `app.listen()` einfügen (OPS-02):**
```js
// Graceful shutdown (OPS-02)
function shutdown(signal) {
    logger.info({ signal }, 'shutdown signal received — closing server');
    server.close(() => {
        logger.info('HTTP server closed');
        closeDb();
        logger.info('database closed — exiting');
        process.exit(0);
    });

    // Force exit after 10 seconds if server.close() hangs
    setTimeout(() => {
        logger.warn('forced exit after timeout');
        process.exit(1);
    }, 10000).unref();
}

process.on('SIGTERM', () => shutdown('SIGTERM'));
process.on('SIGINT', () => shutdown('SIGINT'));
```
